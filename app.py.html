# =========================

@st.cache_data
def load_data():
    return pd.read_csv("golf_data_validated.csv")

df = load_data()

# =========================
# CLEAN DATA
# =========================
df["Position_clean"] = pd.to_numeric(
    df["Position"].astype(str).str.extract(r'(\d+)')[0], errors="coerce"
)
df["score"]           = pd.to_numeric(df["score"],        errors="coerce")
df["Winner Score"]    = pd.to_numeric(df["Winner Score"], errors="coerce")
df["Score_vs_Winner"] = df["score"] - df["Winner Score"]
df = df.dropna(subset=["score", "Winner Score", "Position_clean"])

# =========================
# PAGE HEADER

st.markdown('<div class="page-title">⛳ College Golf Analytics</div>', unsafe_allow_html=True)

    unsafe_allow_html=True,
)

# =========================
# SIDEBAR FILTERS
# =========================
with st.sidebar:
    st.markdown("### 🎛️ Filters")
    st.markdown("---")

    all_teams  = sorted(df["Team"].unique())
    all_events = sorted(df["Event name"].unique())

    teams = st.multiselect("Teams", all_teams, default=all_teams, key="teams_filter")
    events = st.multiselect("Tournaments", all_events, default=all_events, key="events_filter")

    st.markdown("---")
    st.markdown("### 📊 Score Range")
    score_min = int(df["score"].min())
    score_max = int(df["score"].max())
    score_range = st.slider(
        "Filter by total score",
        min_value=score_min,
        max_value=score_max,
        value=(score_min, score_max),
        key="score_range",
    )

    st.markdown("---")
    st.markdown("### ⚔️ Head-to-Head")
    compare_teams = st.multiselect(
        "Select exactly 2 teams",

        key="compare_teams",
    )

    st.markdown("---")
    st.markdown("### 🔢 Display Options")
    top_n = st.selectbox("Top N teams in charts", [5, 10, 15, 20, "All"], index=1, key="top_n")
    show_avg_line = st.checkbox("Show field average line", value=True, key="avg_line")

# =========================
# APPLY FILTERS
# =========================
filtered_df = df[
    df["Team"].isin(teams) &
    df["Event name"].isin(events) &
    df["score"].between(score_range[0], score_range[1])
]

    avg_gap    = filtered_df["Score_vs_Winner"].mean()
    num_events = filtered_df["Event name"].nunique()
    num_teams  = filtered_df["Team"].nunique()
    best_team  = filtered_df.groupby("Team")["Score_vs_Winner"].mean().idxmin()
else:
    best_pos   = "—"
    avg_gap    = None
    num_events = 0

    best_team  = "—"

k1.metric("Best Finish",        f"T{best_pos}" if best_pos != "—" else "—")
k2.metric("Avg Strokes Behind", f"+{avg_gap:.1f}" if avg_gap is not None else "—")
k3.metric("Tournaments",        num_events)
k4.metric("Teams",              num_teams)
k5.metric("Top Performer",      best_team)

# =========================
# HELPER: top-N filter
# =========================
def apply_top_n(frame, col="Score_vs_Winner"):
    if top_n == "All":
        return frame
    top = (
        frame.groupby("Team")[col]
        .mean()
        .nsmallest(int(top_n))
        .index.tolist()
    )
    return frame[frame["Team"].isin(top)]

        pivot = (

            .unstack("Team")
            .dropna()
            .reset_index()
        )

        fig_h2h = go.Figure()
        for i, team in enumerate(compare_teams):
            if team in pivot.columns:
                fig_h2h.add_trace(go.Bar(
                    name=team,
                    x=pivot["Event name"],
                    y=pivot[team],
                    marker_color=colors_h2h[i],
                    opacity=0.85,

            line_dash="dot",
            line_color="rgba(46,204,135,0.4)",
            annotation_text="Winner",
            annotation_font_color="#2ECC87",

        fig_h2h.update_layout(
            **PLOTLY_LAYOUT,
            barmode="group",
            title=f"{t1}  vs  {t2} — strokes behind winner per tournament",

        st.plotly_chart(fig_h2h, use_container_width=True)

        if t1 in pivot.columns and t2 in pivot.columns:
            t1_wins = int((pivot[t1] < pivot[t2]).sum())
            t2_wins = int((pivot[t2] < pivot[t1]).sum())
            ties    = int((pivot[t1] == pivot[t2]).sum())

            col_c1, col_c2, col_c3 = st.columns(3)
            col_c1.metric(f"{t1} events won", t1_wins)
            col_c2.metric("Ties", ties)
            col_c3.metric(f"{t2} events won", t2_wins)

            )

            fig_scatter.update_xaxes(tickangle=-35)
            st.plotly_chart(fig_scatter, use_container_width=True)

            avg_t1  = df_cmp[df_cmp["Team"] == t1]["Score_vs_Winner"].mean()
            avg_t2  = df_cmp[df_cmp["Team"] == t2]["Score_vs_Winner"].mean()

            margin  = abs(avg_t1 - avg_t2)

                unsafe_allow_html=True,
