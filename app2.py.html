import streamlit as st
import pandas as pd
import plotly.express as px

# =========================
# CONFIG
# =========================
st.set_page_config(page_title="Golf Dashboard", layout="wide")

# =========================
# LOAD DATA
# =========================
@st.cache_data
def load_data():
    return pd.read_csv("golf_data_validated.csv")

df = load_data()

# =========================
# CLEAN DATA
# =========================
df["Position_clean"] = df["Position"].astype(str).str.extract(r'(\d+)')
df["Position_clean"] = pd.to_numeric(df["Position_clean"], errors='coerce')

df["score"] = pd.to_numeric(df["score"], errors='coerce')
df["Winner Score"] = pd.to_numeric(df["Winner Score"], errors='coerce')

df["Score_vs_Winner"] = df["score"] - df["Winner Score"]

df = df.dropna(subset=["score", "Winner Score", "Position_clean"])

# =========================
# HEADER
# =========================
st.title("⛳ College Golf Performance Dashboard")
st.markdown("Analyze team performance by tournament and compare against winners.")

# =========================
# SIDEBAR FILTERS
# =========================
st.sidebar.header("Filters")

teams = st.sidebar.multiselect(
    "Select Teams",
    sorted(df["Team"].unique()),
    default=sorted(df["Team"].unique())
)

events = st.sidebar.multiselect(
    "Select Tournaments",
    sorted(df["Event name"].unique()),
    default=sorted(df["Event name"].unique())
)

compare_teams = st.sidebar.multiselect(
    "Compare 2 Teams",
    sorted(df["Team"].unique())
)

# =========================
# FILTER DATA
# =========================
filtered_df = df[
    (df["Team"].isin(teams)) &
    (df["Event name"].isin(events))
]

# =========================
# TABS
# =========================
tab1, tab2, tab3, tab4, tab5 = st.tabs([
    "📍 Map",
    "🏆 Tournament Performance",
    "📊 Score vs Winner",
    "🏫 Leaderboard",
    "⚔️ Comparison"
])

# =========================
# MAP
# =========================
with tab1:
    st.subheader("Tournament Locations")

    if {"Latitude", "Longitude"}.issubset(filtered_df.columns):

        map_df = filtered_df.drop_duplicates(subset=["Event name"])

        fig = px.scatter_mapbox(
            map_df,
            lat="Latitude",
            lon="Longitude",
            hover_name="Event name",
            zoom=3,
            height=500
        )

        fig.update_layout(mapbox_style="open-street-map")

        st.plotly_chart(fig, use_container_width=True)

    else:
        st.warning("Latitude/Longitude not found in dataset.")

# =========================
# TOURNAMENT PERFORMANCE
# =========================
with tab2:
    st.subheader("Average Score by Team per Tournament")

    tournament_team = filtered_df.groupby(
        ["Event name", "Team"]
    )["score"].mean().reset_index()

    fig = px.bar(
        tournament_team,
        x="Event name",
        y="score",
        color="Team",
        barmode="group",
        title="Team Scoring Average by Tournament"
    )

    st.plotly_chart(fig, use_container_width=True)

# =========================
# SCORE VS WINNER (KEY)
# =========================
with tab3:
    st.subheader("Strokes Behind Winner by Tournament")

    diff_df = filtered_df.groupby(
        ["Event name", "Team"]
    )["Score_vs_Winner"].mean().reset_index()

    fig = px.bar(
        diff_df,
        x="Event name",
        y="Score_vs_Winner",
        color="Team",
        barmode="group",
        title="Average Strokes Behind Winner (Per Tournament)"
    )

    st.plotly_chart(fig, use_container_width=True)

    st.markdown("Lower values indicate performance closer to the winner.")

# =========================
# LEADERBOARD (PER EVENT)
# =========================
with tab4:
    st.subheader("Tournament Leaderboard")

    selected_event = st.selectbox(
        "Select Tournament",
        sorted(filtered_df["Event name"].unique())
    )

    event_df = filtered_df[filtered_df["Event name"] == selected_event]

    leaderboard = event_df.groupby("Team").agg({
        "score": "mean",
        "Score_vs_Winner": "mean",
        "Position_clean": "mean"
    }).reset_index()

    leaderboard = leaderboard.sort_values("Score_vs_Winner")

    st.dataframe(leaderboard)

    fig = px.bar(
        leaderboard,
        x="Team",
        y="Score_vs_Winner",
        color="Team",
        title=f"{selected_event} - Team Performance vs Winner"
    )

    st.plotly_chart(fig, use_container_width=True)

# =========================
# TEAM COMPARISON
# =========================
with tab5:
    st.subheader("Head-to-Head Comparison by Tournament")

    if len(compare_teams) != 2:
        st.warning("Select exactly 2 teams.")
    else:
        df_compare = filtered_df[
            filtered_df["Team"].isin(compare_teams)
        ]

        pivot = df_compare.pivot_table(
            index="Event name",
            columns="Team",
            values="Score_vs_Winner"
        )

        st.markdown("### Performance vs Winner per Tournament")
        st.bar_chart(pivot)

        # Insight
        avg = df_compare.groupby("Team")["Score_vs_Winner"].mean()
        better_team = avg.idxmin()

        st.success(f"{better_team} performs closer to tournament winners on average.")

        st.markdown("### Detailed Data")
        st.dataframe(df_compare.sort_values("Event name"))
